Introduction

This document provides a comprehensive derivation and simulation of incompressible duct flow. We explore the transition from inviscid potential theory to the viscous Navier-Stokes equations, focusing on the mechanics of acceleration and pressure recovery.

§1 The Geometry

The duct follows a contraction–throat–diffuser profile defined by a smooth cosine contraction and a linear expansion.

Code 
import numpy as np
import matplotlib.pyplot as plt
import plotly.graph_objects as go
from scipy.integrate import solve_ivp

def duct_h(x, h_in=0.36, h_t=0.14, angle_deg=8.0):
    x_c, x_t = 0.35, 0.45
    alpha = np.tan(np.radians(angle_deg))
    x = np.atleast_1d(x)
    h = np.empty_like(x)
    h[x < x_c] = h_in + (h_t - h_in) * 0.5 * (1 - np.cos(np.pi * x[x < x_c] / x_c))
    h[(x >= x_c) & (x < x_t)] = h_t
    h[x >= x_t] = h_t + alpha * (x[x >= x_t] - x_t)
    return h

x_arr = np.linspace(0, 1, 500)
h_arr = duct_h(x_arr)

plt.figure(figsize=(10, 3), facecolor='#0f1117')
ax = plt.gca()
ax.set_facecolor('#141820')
ax.fill_between(x_arr, -h_arr, h_arr, color='#4a9eff', alpha=0.2)
ax.plot(x_arr, h_arr, color='#8ab0d0', lw=2)
ax.plot(x_arr, -h_arr, color='#8ab0d0', lw=2)
plt.title("Duct Geometry", color='white')
plt.show()

import numpy as np
import matplotlib.pyplot as plt
import matplotlib.patches as mpatches
from matplotlib.collections import LineCollection
import plotly.graph_objects as go
import plotly.express as px
from plotly.subplots import make_subplots

# ── Geometry ──────────────────────────────────────────────────────────────────
def duct_half_height(x, h_in=0.36, h_throat=0.14, angle_deg=8.0):
    """Half-height h(x) for x in [0,1]."""
    x_contract_end = 0.35
    x_throat_end   = 0.45
    alpha = np.tan(np.radians(angle_deg))
    h = np.empty_like(np.atleast_1d(x), dtype=float)
    for i, xi in enumerate(np.atleast_1d(x)):
        if xi < x_contract_end:
            s = xi / x_contract_end
            h[i] = h_in + (h_throat - h_in) * 0.5 * (1 - np.cos(np.pi * s))
        elif xi < x_throat_end:
            h[i] = h_throat
        else:
            h[i] = h_throat + alpha * (xi - x_throat_end)
    return h if h.ndim > 0 else h[0]

x_arr = np.linspace(0, 1, 400)
h_arr = duct_half_height(x_arr)

fig, ax = plt.subplots(figsize=(10, 3))
ax.fill_between(x_arr, h_arr, 0.5, color='#1a2a3a', alpha=0.9)
ax.fill_between(x_arr, -h_arr, -0.5, color='#1a2a3a', alpha=0.9)
ax.plot(x_arr,  h_arr, 'steelblue', lw=2)
ax.plot(x_arr, -h_arr, 'steelblue', lw=2)
for xt, lbl in [(0.17,'Contraction'),(0.40,'Throat'),(0.72,'Diffuser')]:
    ax.text(xt, -0.48, lbl, ha='center', fontsize=9, color='#8ab0d0', style='italic')
ax.set_xlim(0,1); ax.set_ylim(-0.52,0.52)
ax.set_facecolor('#0f1117'); fig.patch.set_facecolor('#0f1117')
ax.tick_params(colors='gray'); ax.set_xlabel('x (normalized)', color='gray')
ax.set_ylabel('y', color='gray')
ax.set_title('Duct geometry: contraction → throat → diffuser', color='#c8d8f0')
plt.tight_layout(); plt.show()

§2 Potential Flow and Bernoulli

In inviscid flow, the speed \bar{U}(x) is inversely proportional to h(x). Bernoulli’s principle states p + \frac{1}{2}\rho U^2 = \text{const}.

§3 The Navier–Stokes Equations

We consider steady, incompressible flow: \rho(\mathbf{u}\cdot\nabla)\mathbf{u} = -\nabla p + \mu\nabla^2\mathbf{u}

In the contraction, convection dominates; in the diffuser, the adverse pressure gradient and viscosity can lead to separation.

§4 Viscosity and the Boundary Layer

Viscosity causes the no-slip condition at the wall, creating parabolic velocity profiles.

§5 Particle Trajectories (Animated)

We integrate the Lagrangian paths \frac{d\mathbf{x}}{dt} = \mathbf{u}.

Code 
import numpy as np
import plotly.graph_objects as go
from scipy.integrate import solve_ivp

def get_vel(x, y, Q=0.08, mode='viscous'):
    h = duct_h(x)[0]
    u_avg = Q / (2*h)
    u = 1.5 * u_avg * (1 - (y/h)**2) if mode=='viscous' else u_avg
    # Approximation for v via continuity
    dx = 1e-4
    dh_dx = (duct_h(x+dx)[0] - h) / dx
    v = -y * (u_avg/h) * dh_dx
    return u, v

n_particles = 9
y0s = np.linspace(-0.25, 0.25, n_particles)
t_eval = np.linspace(0, 2.5, 120)

frames = []
particle_data = [solve_ivp(lambda t, p: get_vel(p[0], p[1]), [0, 2.5], [0, y], t_eval=t_eval).y for y in y0s]

for k in range(len(t_eval)):
    frame_pts = []
    for p in particle_data:
        frame_pts.append(go.Scatter(x=[p[0, k]], y=[p[1, k]], mode='markers', marker=dict(size=6)))
    frames.append(go.Frame(data=frame_pts, name=str(k)))

fig_anim = go.Figure(data=frames[0].data, frames=frames)
fig_anim.update_layout(xaxis=dict(range=[0, 1]), yaxis=dict(range=[-0.5, 0.5]),
                       updatemenus=[dict(type="buttons", buttons=[dict(label="Play", method="animate", args=[None])])],
                       template='plotly_dark')
fig_anim.show()
